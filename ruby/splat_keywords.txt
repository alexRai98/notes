#splat
	def hola(mensaje,*personas)
		personas.each{|persona| puts "#{mensaje} #{persona}"}
	end
	hola("Hey hola","Rai","Carito")
	nombres = ["Rai","Carito",":)",4]
	hola("Hey hola",*nombres) // el splat operator "*" combierte el arreglo en cadenas
	
	Output:
	Hey hola Rai
	Hey hola Carito
	Hey hola :)
	Hey hola 4
